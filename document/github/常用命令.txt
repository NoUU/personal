第一个github相关文件
1.将本地文件夹与本地git相关联
	git init

2.提交文件步骤
	git add 1st_profile.txt
	git commit -m "提交说明1st_profile"

3.查看文件夹文件状态
	git status

4.查看本地文件与gitbug关联状态
	git remote

5.本地git仓库与github关联
	git remote add https://github.com/NoUU/personal.git

6.本地git仓库删除与gitbug关联
	git remote remove origin

7.本地文件推送到github仓库
	git push -u origin master

8.删除本地文件后更新本地git状态
	git -rm 1st_profile.txt
	git commit -m "删除文件"

9.新建文件夹
	mkdir git—test

10.新建文件
	touch 1.txt
